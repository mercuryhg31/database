CREATE TABLE AD_DEPARMENTS
    (
        ID      NUMBER ,
        NAME    VARCHAR2 (50) NOT NULL ,
        HEAD    VARCHAR2 (50) NOT NULL
    );

ALTER TABLE AD_DEPARTMENTS ADD CONSTRAINT DEPARTMENTS_PK PRIMARY KEY (ID);

CREATE TABLE AD_PARENT_INFORMATION
    (
        ID NUMBER ,
        PARENT1_FN  VARCHAR2 (50) NOT NULL ,
        PARENT1_LN  VARCHAR2 (50) NOT NULL ,
        PARENT2_FN  VARCHAR2 (50) ,
        PARENT2_LN  VARCHAR2 (50)
    );

ALTER TABLE AD_PARENT_INFORMATION ADD CONSTRAINT PNT_INFO_PK PRIMARY KEY (ID);
ALTER TABLE AD_PARENT_INFORMATION READ ONLY;

CREATE TABLE AD_STUDENTS
    (
        ID          NUMBER ,
        FIRST_NAME  VARCHAR2 (50) NOT NULL ,
        LAST_NAME   VARCHAR2 (50) NOT NULL ,
        REG_YEAR    DATE NOT NULL ,
        EMAIL       VARCHAR2 (50) NOT NULL ,
        PARENT_ID   NUMBER NOT NULL
    );

ALTER TABLE AD_STUDENTS ADD CONSTRAINT STUDENTS_PK PRIMARY KEY (ID);
ALTER TABLE AD_STUDENTS ADD CONSTRAINT STUDENTS_PARENTS FOREIGN KEY (PARENT_ID) REFERENCES AD_PARENT_INFORMATION (ID);

CREATE TABLE AD_COURSES
    (
        ID  NUMBER ,
        NAME        VARCHAR2 (50) NOT NULL ,
        DEPT_ID     NUMBER ,
        LOGIN_ID    VARCHAR2 (10) ,
        PASSWORD    VARCHAR2 (10) ,
        SESSION_ID  NUMBER NOT NULL ,
        BULDING     VARCHAR2 (50) ,
        ROOM        VARCHAR2 (20) ,
        DATE_TIME   VARCHAR2 (20)
    );

ALTER TABLE AD_COURSES ADD CONSTRAINT COURSES_PK PRIMARY KEY (ID);

CREATE TABLE AD_FACULTY
    (
        ID          NUMBER ,
        FIRST_NAME  VARCHAR2 (50) NOT NULL ,
        LAST_NAME   VARCHAR2 (50) NOT NULL ,
        EMAIL       VARCHAR2 (50) NOT NULL ,
        SALARY      NUMBER ,
        INSURANCE   VARCHAR2 (2) ,
        HOURLY_RATE NUMBER ,
        DEPT_ID     NUMBER
    );

ALTER TABLE AD_FACULTY ADD CONSTRAINT FACULTY_PK PRIMARY KEY (ID);
ALTER TABLE AD_FACULTY ADD CONSTRAINT FACULTY_DEPT FOREIGN KEY (DEPT_ID) REFERENCES AD_DEPARTMENTS (ID);

CREATE TABLE AD_EXAM_TYPES
    (
        TYPE        VARCHAR2 (50) ,
        NAME        VARCHAR2 (50) NOT NULL ,
        DESCRIPTION VARCHAR2 (50)
    );

ALTER TABLE AD_EXAM_TYPES ADD CONSTRAINT EXAM_TP_PK PRIMARY KEY (TYPE);

CREATE TABLE AD_EXAMS
    (
        ID          NUMBER ,
        START_DATE  DATE ,
        EXAM_TYPE   VARCHAR2 (50) NOT NULL ,
        COURSE_ID   NUMBER NOT NULL
    );

ALTER TABLE AD_EXAMS ADD CONSTRAINT EXAMS_PK PRIMARY KEY (ID);
ALTER TABLE AD_EXAMS ADD CONSTRAINT EXAM_TP FOREIGN KEY (EXAM_TYPE) REFERENCES AD_EXAM_TYPES (TYPE);
ALTER TABLE AD_EXAMS ADD CONSTRAINT EXAM_COURSE FOREIGN KEY (COURSE_ID) REFERENCES AD_COURSES (ID);

CREATE TABLE AD_EXAM_RESULTS
    (
        STUDENT_ID  NUMBER ,
        COURSE_ID   NUMBER ,
        EXAM_ID     NUMBER ,
        EXAM_GRADE  NUMBER NOT NULL
    );

CREATE TABLE AD_STUDENT_ATTENDANCE
    (
        STUDENT_ID      NUMBER ,
        SESSION_ID      NUMBER ,
        NUM_WORK_DAYS   NUMBER NOT NULL ,
        NUM_DAYS_OFF    NUMBER NOT NULL ,
        EXAM_ELIGIBLITY VARCHAR2 (20)

ALTER TABLE AD_STUDENT_ATTENDANCE ADD CONSTRAINT ATTENDANCE_STUDENTS FOREIGN KEY (STUDENT_ID) REFERENCES AD_STUDENTS (ID);
ALTER TABLE AD_STUDENT_ATTENDANCE ADD CONSTRAINT ATTENDANCE_SESSIONS FOREIGN KEY (SESSION_ID) REFERENCES AD_ACADEMIC_SESSIONS (ID);

CREATE TABLE AD_STUDENT_COURSE_DETAILS

ALTER TABLE AD_STUDENT_COURSE_DETAILS ADD CONSTRAINT DETAIL_COURSES FOREIGN KEY (COURSE_ID) REFERENCES AD_COURSES (ID);
